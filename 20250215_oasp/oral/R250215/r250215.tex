          \begin{itemize}\setlength\itemsep{1em}
            \item Rapid solution - 1 day after
            \item Final solution - 12 days after
          \end{itemize}
        \item Ανάλυση υψηλών συχνοτήτων (>1Hz)
          \begin{itemize}\setlength\itemsep{1em}
            \item Δεδομένα HEPOS - 1Hz
